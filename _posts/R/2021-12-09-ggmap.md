# 2. 1개 장소 지도에 표시하기

***

<img width="600" src="https://user-images.githubusercontent.com/55419868/145412749-f473f851-4e06-4e72-839f-49bf25ff0bdc.png">

```
skk <- geocode(enc2utf8("명륜4가")) # 위도와 경도는?
skk # (127, 37.6)

cn <- as.numeric(skk) # skk 를 숫자 벡터로 바꿔줘
cn # 정확한 위도 경도 출력 (126.99974  37.58215)

map1 <- get_googlemap(center = cn,
                      maptype = "roadmap",
                      markers = skk,
                      zoom=15)
ggmap(map1) # map을 지도에서 보여줘
```
위도와 경도를 받아서 지도에 나타냅니다.<br>
`center` -> 위도, 경도 <br>
`markers` -> 지도위에 마크표시를 합니다 <br>
`zoom` -> 지도로 보이는 넓이를 조절합니다 <br>

<br><br>


# 3. 여러 장소 지도에 표시하기

<img width="600" src="https://user-images.githubusercontent.com/55419868/145413337-3519e025-c5bc-4540-9d4a-6cd21de3412a.png">

```
court5 <- c("서울중앙법원", "서울동부법원",
            "서울서부법원","서울남부법원","서울북부법원")
court <- geocode(enc2utf8(court5))
court # 위도와 경도

df_court <- data.frame(name = court5,
                       lon=court$lon,
                       lat = court$lat)
df_court # 데이터프레임 (지역이름, 위도, 경도)

# 지도의 중심점 (5개 법원의 위도 경도 평균치)
cn2 <- c(mean(df_court$lon), mean(df_court$lat))
cn2

map2 <- get_googlemap(center = cn2, # 장소들 위도 경도 평균
                       maptype = "roadmap",
                       markers = court, # 장소들 위도 경도
                       zoom = 11)

ggmap(map2) # 지도로 표시

court_map <- ggmap(map2)
court_map + geom_text(data=df_court, # 데이터프레임 사용
                      aes(x=lon, y=lat), # 위도 경도 기준
                      label=court5, size=5) # 라벨 붙이기
```
